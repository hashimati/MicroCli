package ${pack};

import graphql.kickstart.tools.GraphQLMutationResolver
import graphql.kickstart.tools.GraphQLQueryResolver
import ${domainPackage}.${className}
import ${repoPackage}.${className}Repository

import javax.inject.Inject;
import javax.inject.Singleton;

@Singleton
class ${className}QueryResolver implements GraphQLQueryResolver, GraphQLMutationResolver(private val ${entityName}Repository :${className}Repository) {

    fun find${className}ById(var id: Long): ${className}
    {
        return ${entityName}Repository.findById(id).get();
    }

    fun findAll${className}() : Iterable<${className}>
    {
            return ${entityName}Repository.findAll();
    }
}
