import ${pack}.domains.${className};

import javax.inject.Inject;
import javax.inject.Singleton;

@Singleton
class ${className}QueryResolver implements GraphQLQueryResolver(private val ${entityName}Repository :${className}Repository) {

    fun  find(var id: Long): ${className}
    {
        return ${entityName}Repository.findById(id).get();
    }
}
